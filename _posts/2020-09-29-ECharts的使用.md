## 安装 ECharts

通过 `npm` 安装 ECharts 的 Vue 插件，`npm install echarts --save`,或者`yarn`的`yarn add echarts --save`
以及安装 Charts 的 TS 插件`yarn add @types/echarts`

## 引入 ECharts

在 main.ts 里面引入 ECharts,以及在生命周期`mounted`里面初始化 ECharts.`setOption`是ECharts的样式

```Vue
<script lang="ts">
import { Vue, Component, Prop, Watch } from "vue-property-decorator";
import echarts from "echarts";
@Component
export default class Charts extends Vue {
    mounted(){
        const chart = echarts.init(this.$refs.wrapper as HTMLDivElement);
        chart.setOption(this.options);
    }
    options = {
      xAxis: {
        boundaryGap: false,
        splitLine: {
          show: false
        },
        type: 'category',
        data: //这里我的数据是月份,请自行修改
      },
      yAxis: {
        show: false,
        type: 'value'
      },
      series: [
        {
          data: //这里我的数据对应月份的数据,请自行修改
          itemStyle: {
            normal: {
              label: {
                show: true,
                position: 'top',
              }
            }
          },
          areaStyle: {
            shadowColor: 'rgb(30,29,43)',
            color: {
              type: 'linear',
              x: 0,
              y: 0,
              x2: 0,
              y2: 1,
              colorStops: [{
                offset: 0, color: 'rgba(96,102,120,1)' // 0% 处的颜色
              }, {
                offset: 1, color: 'rgba(96,102,120,0.1)' // 100% 处的颜色
              }],
              global: false // 缺省为 false
            }
          },
          name: '月份',
          type: 'line',
          stack: '总量',
          // 径向渐变，前三个参数分别是圆心 x, y 和半径，取值同线性渐变
          color: '#212943',
          label: {
            normal: {
              show: true,
              position: 'top',
              formatter: function (params: any) {
                if (params.value > 0) {
                  return params.value;
                } else {
                  return '';
                }
              },
            }
          },
        }]
    };
}
</script>
```
